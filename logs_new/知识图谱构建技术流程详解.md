### 3. **文档分块阶段**

#### 3.1 智能分块（create_chunks.py）
```python
def split_file_into_chunks(self):
    chunk_size = int(os.environ.get('TEXT_CHUNK_SIZE', 200))  # tokens
    chunk_overlap = int(os.environ.get('TEXT_CHUNK_OVERLAP', 20))
    text_splitter = TokenTextSplitter(chunk_size=chunk_size, chunk_overlap=chunk_overlap)
    
    chunks = []
    for i, document in enumerate(self.pages):
        chunk = text_splitter.split_documents([document])
        # 保留原始metadata（包括图片信息）
        chunk_metadata = {'page_number': i + 1}
        if 'images' in document.metadata:
            chunk_metadata['images'] = document.metadata['images']
        chunks.append(Document(page_content=chunk.page_content, metadata=chunk_metadata))
```

**分块特点**：
- 基于Token的分割（而不是字符）
- 可配置的重叠大小（保持上下文连续性）
- 保留图片与块的关联信息

```python
def update_embedding_create_vector_index(graph, chunkId_chunkDoc_list, file_name):
    # 1. 加载embedding模型
    embeddings, dimension = load_embedding_model(embedding_model_name)
    
    # 2. 为每个chunk生成向量
    data_for_query = []
    for row in chunkId_chunkDoc_list:
        if isEmbedding.upper() == "TRUE":
            embeddings_arr = embeddings.embed_query(row['chunk_doc'].page_content)
            data_for_query.append({
                "chunkId": row['chunk_id'],
                "embeddings": embeddings_arr
            })
    
    # 3. 创建向量索引
    graph.query("""
    CREATE VECTOR INDEX `vector` if not exists for (c:Chunk) on (c.embedding)
    OPTIONS {indexConfig: {
        `vector.dimensions`: $dimensions,
        `vector.similarity_function`: 'cosine'
    }}
    """, {"dimensions": dimension})
    
    # 4. 批量更新向量属性
    graph.query("""
    UNWIND $data AS row
    MATCH (d:Document {fileName: $fileName})
    MERGE (c:Chunk {id: row.chunkId})
    SET c.embedding = row.embeddings
    MERGE (c)-[:PART_OF]->(d)
    """, params={"fileName": file_name, "data": data_for_query})
```

#### 5.2 知识图谱转换的核心逻辑（LLMGraphTransformer）
```python
class LLMGraphTransformer:
    def __init__(self, llm, allowed_nodes, allowed_relationships, 
                 use_fault_diagnosis_domain=True):
        # 支持函数调用的结构化输出
        self._function_call = use_function_call
        self.use_fault_diagnosis_domain = use_fault_diagnosis_domain
        
        # 根据领域选择不同的Prompt
        if use_fault_diagnosis_domain:
            system_prompt = FAULT_DIAGNOSIS_SYSTEM_PROMPT
            examples = FAULT_DIAGNOSIS_EXAMPLES
        else:
            system_prompt = GENERAL_SYSTEM_PROMPT
            examples = GENERAL_EXAMPLES
    
    def process_response(self, document: Document) -> GraphDocument:
        # 1. 调用LLM提取实体和关系
        raw_schema = self.chain.invoke({"input": document.page_content})
        
        # 2. 解析LLM输出为图结构
        if self._function_call:
            nodes, relationships = _convert_to_graph_document(raw_schema)
        else:
            # 回退到JSON解析
            parsed_json = self.json_repair.loads(raw_schema.content)
            nodes, relationships = self._parse_json_to_graph(parsed_json)
        
        # 3. 严格模式过滤
        if self.strict_mode and (self.allowed_nodes or self.allowed_relationships):
            nodes = [node for node in nodes if node.type.lower() in allowed_nodes]
            relationships = [rel for rel in relationships if rel.type.lower() in allowed_relationships]
        
        return GraphDocument(nodes=nodes, relationships=relationships, source=document)
```

**故障诊断领域特殊处理**：
- 专门的System Prompt，指导模型提取故障诊断相关的实体类型
- 预定义的示例（如：Parameter → EXCEEDS → Threshold）
- 支持特定实体类型：Equipment、Component、Fault、Symptom、Diagnosis、Solution等

```python
def update_KNN_graph(self):
    # 查询相似的chunk（基于向量相似度）
    query = """
    MATCH (c:Chunk)
    WHERE c.embedding IS NOT NULL AND count { (c)-[:SIMILAR]-() } < 5
    CALL db.index.vector.queryNodes('vector', 6, c.embedding) 
    yield node, score
    WHERE node <> c and score >= $score
    MERGE (c)-[rel:SIMILAR]-(node) SET rel.score = score
    """
```
